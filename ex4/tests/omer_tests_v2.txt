/// order_of_eval_with_arrays
array IntArray = int[];
int i := 0;
int j := 0;
IntArray a;

int on() {
    a[i] := 1;
    i := i + 1;
    return a[i - 1];
}

int off() {
    a[j] := 0;
    j := j + 1;
    return a[j - 1];
}

void foo(int a, int b, int c, int d) {
    int e := a + b + c + d;
    PrintInt(e);
}

void main() {
    a := new int[5];

    foo(on(), on(), on(), off());

    PrintInt(a[0]);
    PrintInt(a[1]);
    PrintInt(a[2]);
    PrintInt(a[3]);
    PrintInt(a[4]);
}

*** 3 0 1 1 0 0
